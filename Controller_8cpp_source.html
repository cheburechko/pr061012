92  for (int k = 0; k < 2; k++)
93  {
94 
95  // first look for visible objects, then for hidden

99  objects = world -> getVisibleObjects();
100  }
101  else

103  objects = world -> getHiddenObjects();
104  }
105 
106  ObjectHeap::iterator end = objects -> end();
107  for (ObjectHeap::iterator i = objects -> begin(); i != end; i++)
108  {
109  // check objects health
110  if ((*i) -> getHealthPoints() <= 0 && !(*i) -> isDestroyed())
111  {
112  destroy(*i);
113  }
114 
115  // don't do anything with destored objects
116  if ((*i) -> isDestroyed())
117  {
118  continue;
119  }
120 
121  // get actions to the buffer
122  std::vector<Action>* buf;
123 
124  // creatures are special
125  if ((*i) -> getType() == CREATURE)

127  // place view_area at the object

132  dynamic_cast<Creature*>(*i) -> setObjectsAround(
133  world -> getIndexator() -> getAreaContents(view_area, *i));
134 
135  // check age
136  if (dynamic_cast<Creature*>(*i) -> getAge() >=
137  dynamic_cast<Creature*>(*i) -> getMaxAge())
138  {
139  destroy(*i);
140  continue;
141  }
142  buf = (*i) -> getActions();
143  // creatures can't do more than one action
144  if (buf -> size())
145  {
146  actions.push_back(&(buf -> at(0)));
147  }
148  }
149  else

161  {

163  }
164  }
165  }
166  }
167 
168  // Swap buffers

170  i != hiddenToVisible.end(); i++)
171  {

175  }
176

178  i != visibleToHidden.end(); i++)
179  {
180  world -> getHiddenObjects() -> push(*i);
181  world -> getVisibleObjects() -> remove(*i);
182  world -> getIndexator() -> removeObject(*i);
183  }
184 
185  // Execute actions
186  for (uint i = 0; i < actions.size(); i++)
187  {
188  performers[actions[i] -> getType()] -> perform(*actions[i]);
189  }
190 
191  // Try to create weather.
192  if (weather_steps < 0)
193  {
194  world -> simulateWeather();
195  weather_steps = Random::int_range(WEAT_STEPS_MIN, WEAT_STEPS_MAX) /
196  GEN_WEAT_INTENSITY;
197  }
198  else
199  {
200  weather_steps--;
201  }
202 
203  // Reproduce cows and stuff.
204  if (creature_steps < 0)
205  {
206  creature_steps += GEN_CREATURE_STEPS;
207  world -> simulateCreatures();
208  }
209  else

211  creature_steps--;
212  }
213
