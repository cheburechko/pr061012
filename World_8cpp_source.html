155 void World::genForestAt(double x, double y, int x_trees, int y_trees, const ParamArray& tree_params)
156 {
157  const double interval = GEN_TREE_INTERVAL;
158  const double shift = interval/3;
159 
160  // x_trees and y_trees have to be odd values
161  if(x_trees%2 == 0)x_trees++;
162  if(y_trees%2 == 0)y_trees++;
163 
164  // Probabilities of tree generation at certain coordinates
165  double probs[x_trees][y_trees];
166 
167  int x_center = x_trees/2 + 1;
168  int y_center = y_trees/2 + 1;
169 
170  // Calculating probabilities of tree generation based on distance
171  for(int i = 0; i < x_center; ++i)
172  {
173  for(int j = 0; j < y_center; ++j)
174  {
175  double distance = sqrt( pow(x_center - i, 2) + pow(y_center - j, 2) );
176  probs[i][j] = probs[x_trees - i - 1][y_trees - j - 1] = probs[i][y_trees - j - 1] = probs[x_trees - i - 1][j] = 1.0 - distance*GEN_TREE_PROB_DECAY;
177  }
178  }
179 
180  // Generating trees.
181  for(int i = 0; i < x_trees; ++i)
182  {
183  for(int j = 0; j < y_trees; ++j)
184  {
185  genTreeAt(x + (i-x_center)*interval, y + (j-y_center)*interval, shift, probs[i][j], tree_params);
186  }
187  }
188 
189 }
