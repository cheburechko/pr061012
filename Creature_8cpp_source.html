279 void Creature::updateDanger()
280 {
281  ObjectHeap::const_iterator iter;
282  this -> danger = 0;
283 
284  // Only creatures and weather can be dangerous
285  for(

288  )
289  {
290  Creature* creat = dynamic_cast<Creature*>(*iter);
291  if (this -> getDangerLevel() < creat -> getDangerLevel() + 10)
292  this -> danger += creat -> getDangerLevel();
293  }
294 
295  for(
296  iter = objects_around.begin(WEATHER);
297  iter != objects_around.end(WEATHER); iter++
298  )
299  {
300  Weather* weath = dynamic_cast<Weather*>(*iter);

302  this -> danger += weath -> getDangerLevel();

306 void Creature:: chooseDirectionToEscape()
307 {
308  // Initialize vector of escaping.
309  double global_x = 0;
310  double global_y = 0;
311 
312  // Add all dangerous objects danger levels as vectors
313  // with length equal to object's danger level and
314  // angle equal to direction to the object
315  ObjectHeap::const_iterator iter;
316  for(
317  iter = objects_around.begin(CREATURE);
318  iter != objects_around.end(CREATURE); iter++
319  )
320  {
321  if (this -> getDangerLevel() < (*iter) -> getDangerLevel() + 10)
322  {
323  angle = getCoords().getAngle((*iter) -> getCoords());
324  global_x += (*iter) -> getDangerLevel() * cos(angle);
325  global_y += (*iter) -> getDangerLevel() * sin(angle);
326  }
327  }
328 
329  for(
330  iter = objects_around.begin(WEATHER);
331  iter != objects_around.end(WEATHER); iter++
332  )
333  {
334  if (this -> getDangerLevel() < (*iter) -> getDangerLevel() + 10)
335  {
336  angle = getCoords().getAngle((*iter) -> getCoords());
337  global_x += (*iter) -> getDangerLevel() * cos(this -> angle);
338  global_y += (*iter) -> getDangerLevel() * sin(this -> angle);
339  }
340  }
341 
342  // go to the opposite direction of biggest danger
343  this -> angle = atan(global_y / global_x) + M_PI;
344 }

346 // Go with the given speed
347 void Creature::go(SpeedType speed)
348 {
349  // If we could not move, then reset direction
350  if (prev_action == GO && prev_action_state == FAILED)
351  direction_is_set = false;
352 
353  // if we don't have any aim, go the way we went before
354  if (!aim)
355  {
356  // if there is no direction, then go random
357  // or there is an error in angle
358  if (!direction_is_set || isnan(angle))
359  {
360  direction_is_set = true;
361  angle = Random::double_num(M_PI * 2);
362  }
363  }
364  else
365  {
366  // if we can't go the way we went, or the aim changed,
367  // reset route
368  if (!direction_is_set || aim != goal)
369  {
370  goal = aim;
371  //generate route
372  route = generateRoute(goal);
373 
374  angle = this -> getCoords().getAngle(route.top());
375  direction_is_set = true;
376  }
377 
378  // if we reached the target, face to the next point
379  if (this -> getShape().hitTest(route.top()))
380  {
381  route.pop();
382 
383  // check if we reached the end
384  if (route.empty())
385  {
386  direction_is_set = false;
387  return;
388  }
389 
390  // face to the the next point
391  angle = this -> getCoords().getAngle(route.top());
392  direction_is_set = true;
393  }
394  }
395 
396  // Generate GO action
397  Action act(GO, this);
398  act.addParam<double>("angle", angle);
399  act.addParam<SpeedType>("speed", speed);
400  this -> actions.push_back(act);
401 }
