454 // Evaluates object's danger depending on the distance to it.
455 double Creature::evaluateDanger(const Object * obj)
456 {
457  double view_radius = view_area.getSize() / 2;
458  double distance = getCoords().getDistance(obj -> getCoords());
459  double my_radius = getShape().getSize() / 2;
460  double obj_radius = getShape().getSize() / 2;
461  // Dependent on:
462  // - the ratio of object's danger to ours (== 1 if same)
463  double danger_ratio = double(obj -> getDangerLevel()) / getDangerLevel();
464 
465  // - distance to object (== 1 if we stand right next to the object)
466  double distance_ratio = (my_radius + obj_radius) / distance;
467 
468  // - size of view_area (== 0 if we can't see object,
469  // == 1 if we stand right next to the object)
470  double view_ratio = fmax(0, (view_radius + obj_radius - distance) /
471  (view_radius - my_radius));
472 
473  // - constant factor
474  return pow(danger_ratio * distance_ratio, 2) * view_ratio * CREAT_DANGER_FACTOR;
475 }

477 void Creature::chooseDirectionToEscape()
478 {
479  // Initialize vector of escaping.
480  Vector escape_vector(0, 0);
481 
482  // Add all dangerous objects danger levels as vectors
483  // with length equal to object's danger level and
484  // angle equal to direction to the object
485  ObjectHeap::const_iterator iter;

492  escape_vector += Vector(cos(angle), sin(angle)) * evaluateDanger(*iter);

505 // Go with the given speed
506 void Creature::go(SpeedType speed)
507 {
508  // If we could not move, then reset direction
509  if (prev_action == GO && prev_action_state == FAILED)
510  {
511  direction_is_set = false;
512  // If we have the same route to the same aim, try to go random
513  if (aim != nullptr && aim == goal && last_route_size == route.size())
514  {
515  angle = Random::double_num(M_PI * 2);
516  Action act(GO, this);
517  act.addParam<double>("angle", angle);
518  act.addParam<SpeedType>("speed", speed);
519  this -> actions.push_back(act);
520  return;
521  }
522  }
523 
524  // if we don't have any aim, go the way we went before
525  if (!aim)
526  {
527  // if there is no direction, then go random
528  // or there is an error in angle
529  if (!direction_is_set || isnan(angle))
530  {
531  direction_is_set = true;
532  angle = Random::double_num(M_PI * 2);
533  }
534  }
535  else
536  {
537  // if we can't go the way we went, or the aim changed,
538  // reset route
539  if (!direction_is_set || aim != goal)
540  {
541  goal = aim;
542  //generate route
543  last_route_size = route.size();
544  route = generateRoute();
545  angle = this -> getCoords().getAngle(route.top());
546  direction_is_set = true;
547  }
548 
549  // if we reached the target, face to the next point
550  double my_speed = 0;
551  if (speed == SLOW_SPEED)
552  {
553  my_speed = CREAT_SPEED_SLOW_VALUE;
554  }
555  else
556  {
557  my_speed = CREAT_SPEED_FAST_VALUE;
558  }
559 
560  while (getCoords().getDistance(route.top()) < my_speed)
561  {
562  route.pop();
563 
564  // check if we reached the end
565  if (route.empty())
566  {
567  direction_is_set = false;
568  return;
569  }
570 
571  // face to the the next point
572  angle = this -> getCoords().getAngle(route.top());
573  direction_is_set = true;
574  }
575  }
576 
577  // Generate GO action
578  Action act(GO, this);
579  act.addParam<double>("angle", angle);
580  act.addParam<SpeedType>("speed", speed);
581  this -> actions.push_back(act);
582  if (speed == FAST_SPEED)
583  {
584  decreaseEndurance(1);
585  }
586 }

795  common_steps--;
796  }
797 
798  // Starving

803 
804  // Danger updating
