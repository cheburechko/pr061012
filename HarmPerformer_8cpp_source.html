31 void HarmPerformer::perform(Action& action)

33  // Random))

42 
43  uint count_error = 0;
44  // Check type of actor.

47  action.markAsFailed(OBJ_CANT_HARM);

49  }

52  std::vector<Object*> participants = action.getParticipants();
53 
54  // Get shape influence.
55  Shape env_shape;
56  if (type == CREATURE)
57  {
58  if (participants.size() != 1)
59  {
60  action.markAsFailed(TOO_MANY_PARTICIPANTS);
61  return;
62  }
63 
64  if (participants[0] -> isImmortal())
65  {
66  action.markAsFailed(OBJ_IS_IMMORTAL);
67  return;
68  }
69 
70  Creature* actor_cr = dynamic_cast<Creature*>(actor);
71  env_shape = actor_cr -> getReachArea();
72  env_shape.setCenter(actor_cr -> getCoords());
73 
74  if (env_shape.hitTest(participants[0] -> getShape()))
75  {
76  participants[0] -> damage(harm);
77 
78  // Send message about attack.
79  Message msg(UNDER_ATTACK, actor_cr);
80  participants[0] -> receiveMessage(msg);
81  if
82  (
83  (participants[0] -> getHealthPoints() == 0) &&
84  (participants[0] -> getType() == CREATURE)
85  )
86  {
87  Creature* cr = dynamic_cast<Creature*>(participants[0]);

102  else

107  }

132  }
133  }
134 
135  }
136 
137  // If all objects receive attack, then...
138  if (!count_error)
139  {
140  action.markAsSucceeded();
141  }
142  // If all objects are out of reach, then...

157 uint HarmPerformer::calculateHarm(Object* actor)
158 {
159  ObjectType type = actor -> getType();
160  if (type == CREATURE)
161  {
162  Creature* cr = dynamic_cast<Creature*>(actor);
163  CreatureType subtype = cr -> getSubtype();
164  uint force = cr -> getForce();
165 
166  if (subtype == HUMANOID)
167  {
168  return force * Random::int_range(DMG_HUM_MIN, DMG_HUM_MAX);
169  }
170  else
171  {
172  NonHumanoid* nhum = dynamic_cast<NonHumanoid*>(cr);
173  NonHumanoidType sstype = nhum -> getSubsubtype();
174 
175  if (sstype == DRAGON)
176  {
177  return force * Random::int_range(DMG_NHUM_DRG_MIN, DMG_NHUM_DRG_MAX);
178  }
179  else if (sstype == COW)
180  {
181  return force * Random::int_range(DMG_NHUM_COW_MIN, DMG_NHUM_COW_MAX);
182  }
183  else
184  {
185  return force * Random::int_range(DMG_NHUM_MIN, DMG_NHUM_MAX);
186  }
187  }
188  }
189  else if (type == WEATHER)
190  {
191  Weather* weat = dynamic_cast<Weather*>(actor);
192  WeatherType type = weat -> getSubtype();
193 
194  if (type == HURRICANE)
195  {
196  return Random::int_range(DMG_WEAT_HRCN_MIN, DMG_WEAT_HRCN_MAX);
197  }
198  else if (type == METEOR_SHOWER)
199  {
200  return Random::int_range(DMG_WEAT_MS_MIN, DMG_WEAT_MS_MAX);
201  }
202  else
203  {
204  return Random::int_range(DMG_WEAT_MIN, DMG_WEAT_MAX);
205  }
206  }
207  else
208  {
209  return 0;
210  }
