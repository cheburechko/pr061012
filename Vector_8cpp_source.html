57 Vector Vector::project(Vector pt1, Vector pt2) const
58 {
59  double length = pt1.getDistance(pt2);
60  // if a line is not a line, project straight to the vector
61  if (DoubleComparison::areEqual(length, 0))
62  {
63  return pt1;
64  }
65 
66  Vector vec_line = pt2 - pt1;
67  Vector vec_pt = *this - pt1;
68  double scale = scalarProduct(vec_pt, vec_line) /
69  (pow(vec_line.getX(), 2) + pow(vec_line.getY(), 2));
70  return pt1 + Vector(vec_line.getX() * scale, vec_line.getY() * scale);
71 }
