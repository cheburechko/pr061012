58 Vector Vector::project(const Vector& pt1, const Vector& pt2) const
59 {
60  double length = pt1.getDistance(pt2);
61  // if a line is not a line, project straight to the vector
62  if (DoubleComparison::areEqual(length, 0))
63  {
64  return pt1;
65  }
66  // Triangle ABC: projecting C on AB
67  // AB
68  Vector vec_line = pt2 - pt1;
69  // AC
70  Vector vec_pt = *this - pt1;
71  // (AB, AC) = |AB|*|AC|*cos(AB^AC)
72  // |AC'| = |AC|*cos(AB^AC)
73  // scale * |AB| = |AC'| = (AC, AB) / |AB|
74  double scale = scalarProduct(vec_pt, vec_line) /
75  (pow(vec_line.getX(), 2) + pow(vec_line.getY(), 2));
76 
77  // OC' = OA + AC' = OA + AB * |AC'|/|AB|
78  return pt1 + Vector(vec_line.getX() * scale, vec_line.getY() * scale);
79 }

96 // Gets point's projection's parameter on the line,
97 // parametrized by points zero and one.
98 double Vector::getLineParameter(const Vector& zero, const Vector& one) const
99 {
100  // pr_this = zero + (one - zero) * t
101  // From Creature::project:
102  // pr_this = zero + (one - zero) * (this - zero, one - zero) / |one - zero|
103  // t = (this - zero, one - zero) / |one - zero|
104  double dist = zero.getDistance(one);
105  if (DoubleComparison::areEqual(dist, 0))
106  {
107  return 0;
108  }
109  return scalarProduct(*this - zero, one - zero) / dist;
110 }

112 // Finds the intersection of two lines defined with pairs of points.
113 Vector Vector::getLinesIntersection(const Vector& first_start,
114  const Vector& first_end,
115  const Vector& second_start,
116  const Vector& second_end)
117 {
118  Vector v1 = first_end - first_start;
119  Vector v2 = second_end - second_start;
120  double a1 = v1.getY();
121  double b1 = -v1.getX();
122  double c1 = vectorProduct(first_end, first_start);
123  double a2 = v2.getY();
124  double b2 = -v2.getX();
125  double c2 = vectorProduct(second_end, second_start);
126 
127  // Cramer's rule.
128  Vector result;
129  double det = a1 * b2 - a2 * b1;
130 
131  // No intersection or equal lines
132  if (DoubleComparison::areEqual(det, 0))
133  {
134  result.setX(std::numeric_limits<double>::quiet_NaN());
135  return result;
136  }
137 
138  result.setX(c1 * b2 - c2 * b1);
139  result.setY(a1 * c2 - c1 * a2);
140  result /= det;
141  return result;
142 }
