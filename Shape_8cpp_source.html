134 // checks hittest with a point
135 bool Shape::hitTest (const Vector& point) const
136 {
137  switch(type)
138  {
139  case CIRCLE:
140  // if a distance between a point and a circle center is less than
141  // radius, return true
142  if (DoubleComparison::isLess(center.getDistance(point), size/2))
143  {
144  return true;
145  }
146  break;
147 
148  case SQUARE:
149  // if a point lies nearer than sides of a square, return true
150  if (DoubleComparison::isLess(fabs(center.getY() - point.getY()),
151  size/2) &&
152  DoubleComparison::isLess(fabs(center.getX() - point.getX()),
153  size/2))
154  {
155  return true;
156  }
157  break;
158  }
159  return false;
160 }

162 // check hittest with a segment
163 bool Shape::hitTest (const Vector& first, const Vector& second) const
164 {
165  Vector point;
166  std::vector<Vector> vertices = getVertices();
167  switch(type)
168  {
169  case CIRCLE:
170  point = this -> getCenter().project(first, second);
171 
172  // check if sement line is closer than radius and if
173  // one of the segments end lie inside a shape or
174  // if a segment crosses circle with ends outside
175  return DoubleComparison::isLess(
176  this -> getCenter().getDistanceToLine(first, second),
177  this -> getSize() / 2) &&
178  (this -> hitTest(first) || this -> hitTest(second) ||
179  DoubleComparison::isLess(
180  Vector::scalarProduct(second - point,
181  first - point), 0));
182  break;
183 
184  case SQUARE:
185  // Check if segment ends lie inside the square.
186  if (hitTest(first) || hitTest(second))
187  {
188  return true;
189  }
190 
191  vertices = getVertices();
192  // Check if segment intersects any of sides.
193  for (uint i = 0; i < 4; i++)
194  {
195  point = Vector::getLinesIntersection(first, second,
196  vertices[i], vertices[(i+1) % 4]);
197  if (!isnan(point.getX()))
198  {
199  double t = point.getLineParameter(first, second);
200  if (t > 0 && t < 1)
201  {
202  return true;
203  }
204  }
205  }
206 
207  break;
208  }
209  return false;
210 }

213 bool Shape::hitTest (const Shape& shape) const
214 {
215  switch(type)
216  {
217  case CIRCLE:
218  switch(shape.getType())
219  {
220  // if sum of radiuses is more than distance between centers,
221  // return true
222  case CIRCLE:

224  (size + shape.getSize()) / 2))

234  break;
235 
236  case SQUARE:
237  switch(shape.getType())
238  {
239  case CIRCLE:
240  return squareCircleHitTest(*this, shape);
241  break;
242 
243  // if a corner of one square lies inside another square,
244  // return true (this way won't work for abstract rectangles)
245  case SQUARE:
246  std::vector<Vector> verts = getVertices();

271 // hittest of square and circle
272 bool Shape::squareCircleHitTest(const Shape& square, const Shape& circle) const
273 {
274  Vector lb = square.getLeftBottom();
275  Vector rt = square.getRightTop();
276  Vector lt = Vector(lb.getX(), rt.getY());
277  Vector rb = Vector(rt.getX(), lb.getY());
278 
279  // a square can have circle's center inside
280  // and vice versa
281  if (square.hitTest(circle.getCenter()) ||
282  circle.hitTest(square.getCenter()))
283  {
284  return true;
285  }
286 
287  // cicle can intersect any side of the square
288  if (circle.hitTest(lt, rt) ||
289  circle.hitTest(lb, rb) ||
290  circle.hitTest(lb, lt) ||
291  circle.hitTest(rb, rt))
292  {
293  return true;
294  }
295 
296  return false;

303 // Calculates intersections
304 int Shape::intersect(const Shape& shape)
305 {
306  // copy data
307  Vector this_lb = this -> getLeftBottom();
308  Vector this_rt = this -> getRightTop();
309  Vector shape_lb = shape.getLeftBottom();
310  Vector shape_rt = shape.getRightTop();
311 
312  double x1 = this_lb.getX();
313  double y1 = this_lb.getY();
314  double x2 = this_rt.getX();
315  double y2 = this_rt.getY();
316  double x3 = shape_lb.getX();
317  double y3 = shape_lb.getY();
318  double x4 = shape_rt.getX();
319  double y4 = shape_rt.getY();
320 
321  // check if MBB don't intersect
322  if (x4 < x1 || x3 >= x2 || y4 < y1 || y3 >= y2)
323  return NO_INTERSECTION;
324 
325  int result = 0;
326 
327  // check if there is a part of another shape to the left
328  if (x3 < x1 && x4 > x1)
329  result |= LEFT;
330 
331  // to the right
332  if (x3 <= x2 && x4 >= x2)
333  result |= RIGHT;
334 
335  // to the bottom
336  if (y3 < y1 && y4 > y1)
337  result |= BOTTOM;
338 
339  // to the top
340  if (y3 <= y2 && y4 >= y2)
341  result |= TOP;
342 
343  return result;
