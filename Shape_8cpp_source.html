119  return true;
120  }

122  }

131 // Calculates intersections
132 int Shape::intersect(Shape shape)
133 {
134  // copy data
135  Point this_lb = this -> getLeftBottom();
136  Point this_rt = this -> getRightTop();
137  Point shape_lb = shape.getLeftBottom();
138  Point shape_rt = shape.getRightTop();
139 
140  double x1 = this_lb.getX();
141  double y1 = this_lb.getY();
142  double x2 = this_rt.getX();
143  double y2 = this_rt.getY();
144  double x3 = shape_lb.getX();
145  double y3 = shape_lb.getY();
146  double x4 = shape_rt.getX();
147  double y4 = shape_rt.getY();
148 
149  // check if MBB don't intersect
150  if (x4 < x1 || x3 >= x2 || y4 < y1 || y3 >= y2)
151  return NO_INTERSECTION;
152 
153  int result = 0;
154 
155  // check if there is a part of another shape to the left
156  if (x3 < x1 && x4 > x1)
157  result |= LEFT;
158 
159  // to the right
160  if (x3 <= x2 && x4 >= x2)
161  result |= RIGHT;
162 
163  // to the bottom
164  if (y3 < y1 && y4 > y1)
165  result |= BOTTOM;
166 
167  // to the top
168  if (y3 <= y2 && y4 >= y2)
169  result |= TOP;
170 
171  return result;
172 }
