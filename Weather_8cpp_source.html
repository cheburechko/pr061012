60 std::vector<Action>* Weather::getActions()

64  for (uint i = 0; i < this -> actions.size(); i++)

66  Action act = this -> actions[i];

69  need_to_change_direction = true;

75  if (this -> steps > 0) this -> steps--;
76  if (this -> roam_steps > 0) this -> roam_steps--;
77 
78  // Changing direction (if needed).
79  if (this -> roam_steps == 0 || need_to_change_direction)
80  {
81  this -> roam_steps = WEAT_ROAM_STEPS;
82  this -> direction_angle = Random::double_num(2 * M_PI);
83  }
84 
85  // Meteor shower harms all objects.
86  if (subtype == METEOR_SHOWER)

92  {
93  act.addParticipant(*iter);
94  }
95 
96  this -> actions.push_back(act);

99  else if (subtype == RAIN)
100  {
101  ObjectHeap::const_iterator iter;
102  for

104  iter = covered_objs.begin(RESOURCE);

106  iter++

111  if (type == GRASS || type == TREE)
112  {
113  Action act(REGENERATE_OBJ, this);
114  act.addParticipant(res);
115  act.addParam<uint>("object_index", 0);
116  this -> actions.push_back(act);

118  }
119  }
120  // Hurricane moves all objects.

123  ObjectHeap::const_iterator iter;
