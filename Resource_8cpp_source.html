17 Resource::Resource(ResourceType type, uint res_amount) :
18  Object(RESOURCE, true, false),
19  subtype(type),
20  progress(0),
21  steps_to_reg(0)
22 {
23  this -> setShapeType(SHP_RESOURCE);
24  this -> setDangerLevel(DNGR_RESOURCE);
25 
26  // Initialising environment resource.
27  if (this -> subtype == WATER)
28  {
29  this -> mineable = false;
30  this -> restorable = false;
31  this -> makeNonPickable();
32  this -> makeSolid();
33  this -> difficulty = RES_DEFAULT_DIFFICULTY;
34  this -> amount_per_drop = 0;
35  this -> reg_amount = 0;
36  this -> amount = 1;
37  this -> max_amount = 1;
38  this -> setShapeSize(Random::double_range(SZ_RES_WATER_DIAM_MIN, SZ_RES_WATER_DIAM_MAX));
39  this -> makeImmortal();
40  }
41  // Initialising food resource.
42  else if
43  (
44  this -> subtype == GRASS || this -> subtype == MEAT ||
45  this -> subtype == BERRIES
46  )
47  {
48  this -> mineable = false;
49  this -> restorable = false;
50  this -> makePickable();
51  this -> makeNonSolid();
52  this -> difficulty = RES_DEFAULT_DIFFICULTY;
53  this -> amount_per_drop = 0;
54  this -> reg_amount = 0;
55 
56  switch (this -> subtype)
57  {
58  case GRASS:
59  this -> amount = res_amount != 0 ? res_amount : Random::int_range(RES_GRASS_AMOUNT_MIN, RES_GRASS_AMOUNT_MAX);
60  this -> setShapeSize(Random::double_range(SZ_RES_GRASS_DIAM_MIN, SZ_RES_GRASS_DIAM_MAX));
61  break;
62 
63  case BERRIES:
64  this -> amount = res_amount != 0 ? res_amount : Random::int_range(RES_BERRIES_AMOUNT_MIN, RES_BERRIES_AMOUNT_MAX);
65  this -> setShapeSize(Random::double_range(SZ_RES_BERRIES_DIAM_MIN, SZ_RES_BERRIES_DIAM_MAX));
66  break;
67 
68  case MEAT:
69  this -> amount = res_amount != 0 ? res_amount : Random::int_range(RES_MEAT_AMOUNT_MIN, RES_MEAT_AMOUNT_MAX);
70  this -> setShapeSize(Random::double_range(SZ_RES_MEAT_DIAM_MIN, SZ_RES_MEAT_DIAM_MAX));
71  break;
72 
73  default:
74  break;
75  }
76 
77  this -> max_amount = this -> amount;
78  }
79  // Initialising tree.
80  else if (this -> subtype == TREE)
81  {
82  this -> mineable = true;
83  this -> restorable = true;
84  this -> makeNonPickable();
85  this -> makeSolid();
86  this -> difficulty = RES_TREE_DIFFICULTY;
87  this -> amount = res_amount != 0 ? res_amount : Random::int_range(RES_TREE_AMOUNT_MIN, RES_TREE_AMOUNT_MAX);
88  this -> max_amount = 1.5 * this -> amount;
89  this -> amount_per_drop = Random::int_range(RES_TREE_DROP_MIN, RES_TREE_DROP_MAX);
90  this -> reg_amount = RES_TREE_REG_AMOUNT;
91  this -> setShapeSize(Random::double_range(SZ_RES_TREE_DIAM_MIN, SZ_RES_TREE_DIAM_MAX));
92  }
93  // Unknown resource type. Using defaults.
94  else
95  {
96  this -> mineable = false;
97  this -> restorable = false;
98  this -> difficulty = RES_DEFAULT_DIFFICULTY;
99  this -> amount = res_amount;
100  this -> max_amount = res_amount;
101  this -> amount_per_drop = 0;
102  this -> reg_amount = 0;
103  this -> setShapeSize(0);
104  }
105 
106  this -> setWeight(WGHT_RESOURCE * this -> amount);
107 }

136 uint Resource::damage(uint delta)
137 {
138  uint d = delta;
139 
140  if (this -> amount < d)
141  {
142  d = this -> amount;
143  }
144

180 std::vector<Action>* Resource::getActions()
181 {
182  this -> actions.clear();
183 
184  if(this -> steps_to_reg-- == 0)
