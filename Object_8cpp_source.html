60 std::string Object::printObjectInfo(bool full) const
61 {
62  std::stringstream ss;
63 
64  // Printing information.
65  ss << "#" << id << " (" << getTypeName() << ")" << std::endl <<
66  // Common flags.
67  (destroyed ? "Destroyed\n" : "") <<
68  (immortality ? "Immortal\n" : "") <<
69  (solidity ? "Solid\n" : "") <<
70  (movable ? "Movable\n" : "") <<
71  (pickable ? "Pickable\n" : "");
72 
73  if (flyable)
74  {
75  ss << "Flyable";
76  if (isCurrentlyFlying())
77  {
78  ss << " (currently flying)";
79  }
80  ss << "\n";
81  }
82 
83  // Coordinates.
84  ss << insertSpaces("Center") << "(" << shape.getCenter().getX() <<
85  ", " << shape.getCenter().getY() <<
86  ")" << std::endl <<
87  // Health points.
88  insertSpaces("HP") << getHealthPoints() << "/" <<
89  getMaxHealthPoints() << std::endl <<
90  // Danger level and weight.
91  insertSpaces("Danger level") << danger_level << std::endl <<
92  insertSpaces("Weight") << weight << std::endl;
93 
94  // Information only for full-mode.
95  if (full)
96  {
97  ss << insertSpaces("Speed") << normal_speed << std::endl <<
98  insertSpaces("Shape") << shape.getTypeName() << " (" <<
99  shape.getSize() << ")" << std::endl;
100  }
101 
102  return ss.str();
103 }
