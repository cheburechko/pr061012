39 std::string sformat(const std::string& format, ...)
40 {
41  int size = 100;
42  std::string result;
43  va_list ap;
44 
45  while (true)
46  {
47  result.resize(size);
48 
49  va_start(ap, format);
50  int n = vsnprintf((char *) result.c_str(), size, format.c_str(), ap);
51  va_end(ap);
52 
53  if (n > -1 && n < size)
54  {
55  result.resize(n);
56  break;
57  }
58 
59  if (n > -1)
60  {
61  size = n + 1;
62  }
63  else
64  {
65  size *= 2;
66  }
67  }
68 
69  return result;
70 }

375 std::string CLI::list(std::stringstream& ss)
376 {
377  std::string output;
378 
379  // Reading type.
380  std::string type = "";
381  ss >> type;
382 
383  // Listing visible objects.
384  ObjectHeap* objs = this -> world -> getVisibleObjects();
385 
386  // Preparing begin and end iterators.
387  ObjectHeap::const_iterator iter_begin, iter_end;
388  if (type == "")
389  {
390  iter_begin = objs -> begin();
391  iter_end = objs -> end();
392  }
393  else if (type == TN_BUILDING)
394  {
395  iter_begin = objs -> begin(BUILDING);
396  iter_end = objs -> end(BUILDING);
397  }
398  else if (type == TN_CREATURE)
399  {
400  iter_begin = objs -> begin(CREATURE);
401  iter_end = objs -> end(CREATURE);
402  }
403  else if (type == TN_RESOURCE)
404  {
405  iter_begin = objs -> begin(RESOURCE);
406  iter_end = objs -> end(RESOURCE);
407  }
408  else if (type == TN_WEATHER)
409  {
410  iter_begin = objs -> begin(WEATHER);
411  iter_end = objs -> end(WEATHER);
412  }
413  else
414  {
415  return sformat("Error: unknown ObjectType (%s).\n", type.c_str());
416  }
417 
418  ObjectHeap::const_iterator iter;
419  for (iter = iter_begin; iter != iter_end; iter++)
420  {
421  Object* obj = *iter;
422 
423  // Preparing flags.
424  std::string flags = "v";
425  flags += obj -> isDestroyed() ? "d" : " ";
426  flags += obj -> isImmortal() ? "i" : " ";
427  flags += obj -> isSolid() ? "s" : "";
428 
429  // Printing output.
430  output += sformat("%d\t%s\t%s\t(%f,\t%f)\n", obj -> getObjectID(),
431  flags.c_str(),
432  this -> obj_types[obj -> getType()].c_str(),
433  obj -> getCoords().getX(),
434  obj -> getCoords().getY());
435  }
436 
437  return output;
438 
439  // TODO: Listing hidden objects.

441 }
